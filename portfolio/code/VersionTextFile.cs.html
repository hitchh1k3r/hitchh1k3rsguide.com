using System.IO; 


using System.Collections; 


using System.Collections.Generic; 


using UnityEngine; 


using UnityEditor; 


  


namespace HitchLib 


{ 


  


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*// 


  public class VersionTextFile : IBuildDeployEventListener, IBuildDeployConfig 


  { 


    /* Saves the build version to a configurable text file before every build. 


       Useful for a game that needs to read it's own version. */ 


  


    // Configurable options: 


    private TextAsset versionFile; 


  


    // START INTERFACE IBuildDeployConfig < < < < < < < < < < < < < < < < < < < < < < < < < < < < < 


    public string GetStorageID() 


    { 


      /* Returns name/ID for JSON storage and retreval. */ 


  


      return "Version Text File"; 


    } 


  


    public BuildAndDeployUtils.ConfigOption[] GetOptions() 


    { 


      /* Saves state to options array. */ 


  


      BuildAndDeployUtils.ConfigOption[] options = new BuildAndDeployUtils.ConfigOption[1]; 


      options[0].name = "Version File"; 


      options[0].type = BuildAndDeployUtils.ConfigType.OBJREF; 


      options[0].rType = typeof(TextAsset); 


      options[0].rVal = versionFile; 


  


      return options; 


    } 


  


    public void SetConfig(BuildAndDeployUtils.ConfigOption[] config) 


    { 


      /* Loads state from options array. */ 


  


      foreach(BuildAndDeployUtils.ConfigOption option in config) 


      { 


        if(option.name == "Version File" && option.rVal is TextAsset) 


        { 


          versionFile = (TextAsset) option.rVal; 


        } 


      } 


    } 


    // END INTERFACE IBuildDeployConfig > > > > > > > > > > > > > > > > > > > > > > > > > > > > > > 


  


    // START INTERFACE IBuildDeployEventListener < < < < < < < < < < < < < < < < < < < < < < < < < < 


    public void PreBuild(BuildDef build, BuildAndDeploySettings settings) 


    { 


      /* Writes the current build version to a TextAsset. */ 


  


      if(versionFile != null) 


      { 


        File.WriteAllText(AssetDatabase.GetAssetPath(versionFile.GetInstanceID()), build.version); 


      } 


    } 


  


    public void PostBuild(BuildDef build, BuildAndDeploySettings settings) 


    {} 


  


    public void CancelBuild(BuildDef build, BuildAndDeploySettings settings) 


    {} 


  


    public void PreDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings) 


    {} 


  


    public void PostDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings) 


    {} 


  


    public void CancelDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings) 


    {} 


    // END INTERFACE IBuildDeployEventListener > > > > > > > > > > > > > > > > > > > > > > > > > > > 


  


  } 


  


}
