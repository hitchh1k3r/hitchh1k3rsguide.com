using System.Collections;


using System.Collections.Generic;


using UnityEngine;


 


public class Notifications : MonoBehaviour


{


 


  // Settings:


  public SpriteEntry[] heroes;


  public GameObject notification;


  public Sprite progressSprite;


 


  // Cache/State:


  private List<GameObject> notifications = new List<GameObject>();


  private Dictionary<int, Sprite> heroeSprites = new Dictionary<int, Sprite>();


 


  // Instance:


  private static Notifications instance;


 


  // MESSAGES //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


 


  void Awake()


  {


    instance = this;


 


    // construct our dictionary


    foreach(SpriteEntry entry in heroes)


    {


      heroeSprites.Add(entry.heroID, entry.sprite);


    }


  }


 


  // INTERNAL //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


 


  private void AddNotification(int hero, Sprite item, string text)


  {


    if(string.IsNullOrEmpty(text))


    {


      // invalid notification...


      return;


    }


 


    if(notifications.Count > 2)


    {


      // move all previous notifications to make room for the new one


      int i = 3-notifications.Count;


      for(int q = 0; q < notifications.Count; ++q)


      {


        GameObject go = notifications[q];


        if(i <= 0)


        {


          StartCoroutine(Tween.MoveTo(go.transform, new Vector3(-125, -55, 0), 0.85f,


                RemoveNotification, Tween.EASE_QUAD_INOUT));


          foreach(UnityEngine.UI.Graphic graphic in


                go.GetComponentsInChildren<UnityEngine.UI.Graphic>())


          {


            StartCoroutine(Tween.FadeOut(graphic, 0.8f, easeType: Tween.EASE_QUAD_INOUT));


          }


        }


        StartCoroutine(Tween.MoveTo(go.transform, new Vector3(-125, 55 - (i*105), 0), 0.8f,


              easeType: Tween.EASE_QUAD_INOUT));


        ++i;


      }


    }


 


    // build the new notification


    GameObject nGo = Instantiate(notification);


    nGo.transform.SetParent(transform, false);


    ((RectTransform) nGo.transform).SetInsetAndSizeFromParentEdge(RectTransform.Edge.Top,


          (Mathf.Min(notifications.Count, 2) * 105), 100);


    Vector3 offset = new Vector3(((RectTransform) nGo.transform).rect.width + 5, 0, 0);


    nGo.transform.localPosition += offset;


 


    // NOTE DANGER ZONE!


    //   here we are doing lazy access to the children elements, adding them to an array, and then


    //   accessing them by the index we know (now) that they will have. In a second pass clean this


    //   up and pass the data into a NotificationElement Component to apply the text and icons.


    UnityEngine.UI.Text[] textComps = nGo.GetComponentsInChildren<UnityEngine.UI.Text>();


    UnityEngine.UI.Image[] icons = nGo.GetComponentsInChildren<UnityEngine.UI.Image>();


    textComps[0].text = text;


    textComps[1].text = "@" + CharacterProfile.GetHeroHandle(hero - 1);


    if(heroeSprites.ContainsKey(hero))


    {


      icons[1].sprite = heroeSprites[hero];


    }


    else


    {


      icons[1].enabled = false;


    }


    if(item != null)


    {


      icons[2].sprite = item;


    }


    else


    {


      icons[2].enabled = false;


    }


 


    notifications.Add(nGo);


 


    // show it


    StartCoroutine(Tween.MoveTo(nGo.transform, nGo.transform.localPosition - offset, 1f,


          easeType: Tween.EASE_PRINT));


    foreach(UnityEngine.UI.Graphic graphic in nGo.GetComponentsInChildren<UnityEngine.UI.Graphic>())


    {


      StartCoroutine(Tween.FadeIn(graphic, 0.8f, easeType: Tween.EASE_QUAD_INOUT));


    }


  }


 


  private void RemoveNotification()


  {


    Destroy(notifications[0]);


    notifications.RemoveAt(0);


  }


 


  // STATIC INTERFACE //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*


 


  public static void Notify(int hero, Sprite item, string text)


  {


    instance.AddNotification(hero, item, text);


  }


 


  public static Sprite ProgressSprite()


  {


    return instance.progressSprite;


  }


 


  public static Sprite GetFace(int heroID)


  {


    return instance.heroeSprites[heroID];


  }


 


  //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


 


  [System.Serializable]


  public struct SpriteEntry


  {


    // A Unity Editor serializable dictionary pair... I ought to make a dictionary entry Editor some


    //   day ^_^ but here we are using heroID as a key, and using that to point to a Sprite value!


    public int heroID;


    public Sprite sprite;


  }


 


}
