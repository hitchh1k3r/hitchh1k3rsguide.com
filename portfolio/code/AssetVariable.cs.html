using System;


using UnityEngine;


 


namespace HitchLib


{


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  public abstract class AssetVariable : ScriptableObject


  {


 


    /* An abstract class to allow the AssetVariableReverter to search for all assetvariables! */


 


#if UNITY_EDITOR


    public abstract void stashValue();


    public abstract void unstashValue();


 


    public virtual bool inspectorEditable()


    {


      return true;


    }


#endif


 


  }


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  public abstract class AssetVariable<TPT> : AssetVariable


  {


 


    /* An abstract template class that each type of assetvariable can extend. It is simple to


       provide a new type:


          [CreateAssetMenu(menuName = "Variable/Float")]


          public class FloatVariable : AssetVariable<float> {}


       And if you want a pre-set/variable style type:


          [Serializable]


          public class FloatReference : AssetReference<FloatVariable, float> {}


       You could even do this (for a asset referanced enum variable):


          public enum TimeOfDay { DAY, NIGHT }


          [CreateAssetMenu(menuName = "Variable/Time Of Day")]


          public class TimeOfDayVariable : HitchLib.AssetVariable<TimeOfDay> {}


          [Serializable]


          public class TimeOfDayReference : HitchLib.AssetReference<TimeOfDayVariable, TimeOfDay> {}


    */


 


    public TPT value;


 


    public void SetValue(TPT value)


    {


      this.value = value;


    }


 


#if UNITY_EDITOR


    private TPT saveValue;


 


    public override void stashValue()


    {


      saveValue = value;


    }


 


    public override void unstashValue()


    {


      value = saveValue;


    }


#endif


 


  }


 


#if UNITY_EDITOR


  // HitchLib // AssetVariable //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*/


  [UnityEditor.InitializeOnLoadAttribute]


  public static class AssetVariableReverter


  {


 


    /* Because GameObject's serialized properties revert their state when the editor exits


       playmode, it make sense that our variable assets would do the same... but unity does not


       revert assets when playmode exists, so we have our own system. */


 


    static AssetVariableReverter()


    {


      UnityEditor.EditorApplication.playModeStateChanged += Reverter;


    }


 


    private static void Reverter(UnityEditor.PlayModeStateChange state)


    {


      if(state == UnityEditor.PlayModeStateChange.EnteredPlayMode)


      {


        foreach(string guid in UnityEditor.AssetDatabase.FindAssets("t:AssetVariable"))


        {


          string path = UnityEditor.AssetDatabase.GUIDToAssetPath(guid);


          UnityEditor.AssetDatabase.LoadAssetAtPath<AssetVariable>(path).stashValue();


        }


      }


      else if(state == UnityEditor.PlayModeStateChange.ExitingPlayMode)


      {


        foreach(string guid in UnityEditor.AssetDatabase.FindAssets("t:AssetVariable"))


        {


          string path = UnityEditor.AssetDatabase.GUIDToAssetPath(guid);


          UnityEditor.AssetDatabase.LoadAssetAtPath<AssetVariable>(path).unstashValue();


        }


      }


    }


  }


#endif


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  [Serializable]


  public abstract class AssetReference


  {}


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  [Serializable]


  public abstract class AssetReference<TAV, TPT> : AssetReference where TAV: AssetVariable<TPT>


  {


 


    /* Also see AssetVariable<TPT> to learn about this. An AssetReference will provide a value to


       any scripts that accesses it. The value can either be hard coded, or a referance to an


       assetvariable of the compatable type. */


 


    // Public State:


    public bool useConstant = true;


    public TPT constantValue;


    public TAV variable;


 


    // Accessors:


    public TPT value


    {


      get { return useConstant ? constantValue : variable.value; }


    }


 


    public static implicit operator TPT(AssetReference<TAV, TPT> reference)


    {


      return reference.value;


    }


 


    // Constructors:


 


    public AssetReference() {}


 


    public AssetReference(TPT value)


    {


      this.useConstant = true;


      this.constantValue = value;


    }


 


    public AssetReference(TAV variable)


    {


      this.useConstant = false;


      this.variable = variable;


    }


 


  }


 


}
