using System.Collections.Generic;


using UnityEngine;


using UnityEditor;


 


namespace HitchLib


{


 


  public class ScriptableSubObjectList : EditorGUIExtension.ReorderableList


  {


 


    // Instance Configuration:


    private bool selectionInspection; // show the selected items inspector under the list


    private bool hideSubAssets; // do not show subassets in the project view window


 


    // State:


    private Editor cachedEditor;


    private Object lastAsset;


    private Object currentObj;


    private bool foldoutSubEditor = true;


 


    //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*


 


    public ScriptableSubObjectList(string title, SerializedProperty property,


          IEnumerable<System.Type> types, bool selectionInspection = true,


          bool renameableItems = false, bool hideSubAssets = false,


          bool bTitle = false) : base(property)


    {


      this.selectionInspection = selectionInspection;


      this.hideSubAssets = hideSubAssets;


 


      // reorderable list overrides:


      reorderableList.drawHeaderCallback = rect =>


            {


              EditorGUI.LabelField(rect, title, bTitle ? EditorStyles.bLabel :


                    EditorStyles.label);


            };


 


      reorderableList.drawElementCallback = (rect, index, isActive, isFocused) =>


            {


              SerializedProperty element = property.GetArrayElementAtIndex(index);


              Object item = element.objectReferenceValue;


              if (item != null)


              {


                rect.y += 2;


                if (renameableItems)


                {


                  EditorGUI.BeginChangeCheck();


                  item.name = EditorGUI.DelayedTextField(new Rect(rect.x + 10, rect.y,


                        rect.width - 20, EditorGUIUtility.singleLineHeight), item.name);


                  if (EditorGUI.EndChangeCheck())


                  {


                    AssetDatabase.ImportAsset(AssetDatabase.GetAssetPath(item));


                  }


                }


                else


                {


                  EditorGUI.LabelField(new Rect(rect.x, rect.y, rect.width,


                        EditorGUIUtility.singleLineHeight), item.name);


                }


              }


            };


 


      reorderableList.onRemoveCallback = list =>


            {


              SerializedProperty element = property.GetArrayElementAtIndex(list.index);


              Object item = element.objectReferenceValue;


              element.objectReferenceValue = null;


              Object.DestroyImmediate(item, true);


              UnityEditorInternal.ReorderableList.defaultBehaviours.DoRemoveButton(list);


              AssetDatabase.ImportAsset(


                    AssetDatabase.GetAssetPath(property.serializedObject.targetObject));


            };


 


      reorderableList.onAddDropdownCallback = (buttonRect, list) =>


            {


              GenericMenu menu = new GenericMenu();


              foreach (System.Type type in types)


              {


                ClickData data = new ClickData();


                data.type = type;


                menu.AddItem(new GUIContent(type.Name), false, ClickHandler, data);


              }


              menu.ShowAsContext();


            };


    }


 


    // INTERFACE //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


 


    public void CleanUp()


    {


      /* Cleanup must be called by the editor that instantiated this object


         (ie. in the Editor's OnDestroy method). */


 


      if (cachedEditor != null)


      {


        Object.DestroyImmediate(cachedEditor);


      }


    }


 


    public void Draw()


    {


      /* Draws this list into the Editor. */


 


      // draw the list:


      Draw(!property.serializedObject.isEditingMultipleObjects);


 


      if (selectionInspection && !property.serializedObject.isEditingMultipleObjects)


      {


        // draw the selected assets inspector under the list:


        if (reorderableList.index >= 0)


        {


          currentObj = property.GetArrayElementAtIndex(reorderableList.index).objectReferenceValue;


        }


        else


        {


          currentObj = null;


        }


 


        if (currentObj != lastAsset)


        {


          // update the cached editor:


          lastAsset = currentObj;


          if (currentObj != null)


          {


            if (cachedEditor != null)


            {


              Object.DestroyImmediate(cachedEditor);


            }


            Editor.CreateCachedEditor(currentObj, null, ref cachedEditor);


          }


        }


 


        if (cachedEditor != null)


        {


          if (currentObj == null)


          {


            Object.DestroyImmediate(cachedEditor);


            cachedEditor = null;


          }


          else


          {


            // draw the cached editor:


            EditorGUILayout.Space();


            EditorGUILayout.Space();


            Rect titleRect = EditorGUILayout.GetControlRect(false,


                  EditorGUIUtility.singleLineHeight);


            titleRect.x = 0;


            titleRect.width = EditorGUIUtility.currentViewWidth;


            foldoutSubEditor = EditorGUI.InspectorTitlebar(titleRect, foldoutSubEditor,


                  cachedEditor.target, true);


            if (foldoutSubEditor)


            {


              EditorGUILayout.Space();


              EditorGUILayout.Space();


              cachedEditor.OnInspectorGUI();


            }


          }


        }


      }


    }


 


    // INTERNAL //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*


 


    private void ClickHandler(object data)


    {


      /* Called when a subobject type is selected from the add list. This creates a new asset as a


         subasset of the list asset, of the type passed in through the data parameter. */


 


      ClickData clickData = (ClickData) data;


 


      Object item = ScriptableObject.CreateInstance(clickData.type);


      item.name = clickData.type.Name;


      if (hideSubAssets)


      {


        item.hideFlags = HideFlags.HideInHierarchy;


      }


      AssetDatabase.AddObjectToAsset(item, property.serializedObject.targetObject);


      AssetDatabase.ImportAsset(AssetDatabase.GetAssetPath(item));


 


      int index = property.arraySize;


      property.arraySize++;


      reorderableList.index = index;


      SerializedProperty element = property.GetArrayElementAtIndex(index);


      element.objectReferenceValue = item;


      currentObj = item;


      property.serializedObject.ApplyModifiedProperties();


    }


 


    //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*


 


    private struct ClickData


    {


      public System.Type type;


    }


 


  }


 


}
