float vector2_dot(Vector2 lhv, Vector2 rhv)


{

Vector3 vector3_cross(Vector3 lhv, Vector3 rhv)


{


  Vector3 r = {{lhv.value.y * rhv.value.z - lhv.value.z * rhv.value.y,


                lhv.value.z * rhv.value.x - lhv.value.x * rhv.value.z,


                lhv.value.x * rhv.value.y - lhv.value.y * rhv.value.x}};


  return r;


}


 


 


Vector2 vector2_rotate(float theta, Vector2 v)


{


  float cosine = cosf(theta);


  float sine = sinf(theta);


  Vector2 r = {{v.value.x * cosine - v.value.y * sine,


                v.value.y * cosine + v.value.x * sine}};


  return r;


}


 


Vector3 vector3_rotate(Quaternion q, Vector3 v)


{


  float x2 = 2 * q.value.x;


  float y2 = 2 * q.value.y;


  float z2 = 2 * q.value.z;


  float xx2 = q.value.x * x2;


  float yy2 = q.value.y * y2;


  float zz2 = q.value.z * z2;


  float xy2 = q.value.x * y2;


  float xz2 = q.value.x * z2;


  float yz2 = q.value.y * z2;


  float wx2 = q.value.w * x2;


  float wy2 = q.value.w * y2;


  float wz2 = q.value.w * z2;


 


  Vector3 r = {{(1 - (yy2 + zz2)) * v.value.x + (xy2 - wz2) * v.value.y + (xz2 + wy2) * v.value.z,


                (xy2 + wz2) * v.value.x + (1 - (xx2 + zz2)) * v.value.y + (yz2 - wx2) * v.value.z,


                (xz2 - wy2) * v.value.x + (yz2 + wx2) * v.value.y + (1 - (xx2 + yy2)) * v.value.z}};


  return r;


}


 


 


Vector3 vector3_transform(Matrix4 m, Vector3 v) // FORWARD DECLARED


{


  float scalar = 1.0f / (m.array[3] * v.value.x + m.array[7] * v.value.y +


                         m.array[11] * v.value.z + m.array[15]);


  Vector3 r = {{scalar * (m.array[0] * v.value.x + m.array[4] * v.value.y +


                          m.array[8] * v.value.z + m.array[12]),


                scalar * (m.array[1] * v.value.x + m.array[5] * v.value.y +


                          m.array[9] * v.value.z + m.array[13]),


                scalar * (m.array[2] * v.value.x + m.array[6] * v.value.y +


                          m.array[10] * v.value.z + m.array[14])}};


  return r;


}

Quaternion quaternion_fromEuler(float x, float y, float z)


{


  // The x, y, z input might be wrong!


  float cx = cosf(x * 0.5f);


  float sx = sinf(x * 0.5f);


  float cy = cosf(y * 0.5f);


  float sy = sinf(y * 0.5f);


  float cz = cosf(z * 0.5f);


  float sz = sinf(z * 0.5f);


 


  Quaternion r = {{cx * sz * cy - sx * cz * sy,


                   cx * cz * sy + sx * sz * cy,


                   sx * cz * cy - cx * sz * sy,


                   cx * cz * cy + sx * sz * sy}};


  return r;


}


 


Vector3 quaternion_toEuler(Quaternion q)


{


  Vector3 r = {{}};


  return r;


}


 


Quaternion quaternion_inverse(Quaternion q)


{


  Quaternion r = {{-q.value.x, -q.value.y, -q.value.z, q.value.w}};


  return r;


}


 


Quaternion quaternion_multiply(Quaternion lhq, Quaternion rhq)


{


  // NOTE x, y, z, w might need to be y, z, w, x


  // and the w line might need to be the first (x) line!


  Quaternion r = {{rhq.value.w*lhq.value.x + rhq.value.x*lhq.value.w -


                   rhq.value.y*lhq.value.z + rhq.value.z*lhq.value.y,


                   rhq.value.w*lhq.value.y + rhq.value.x*lhq.value.z +


                   rhq.value.y*lhq.value.w - rhq.value.z*lhq.value.x,


                   rhq.value.w*lhq.value.z - rhq.value.x*lhq.value.y +


                   rhq.value.y*lhq.value.x + rhq.value.z*lhq.value.w,


                   rhq.value.w*lhq.value.w - rhq.value.x*lhq.value.x -


                   rhq.value.y*lhq.value.y - rhq.value.z*lhq.value.x}};


  return r;


}

Matrix4 matrix4_angleAxis(float angle, float x, float y, float z)


{


  // FROM https://github.com/hitchh1k3r/gitFighter/blob/gh-pages/web/src/VectorMath.dart


  angle = angle * PI / 180.0;


  float invLen = 1 / sqrtf(x*x + y*y + z*z);


  x *= invLen;


  y *= invLen;


  z *= invLen;


 


  float s = sinf(angle);


  float c = cosf(angle);


  float t = 1 - c;


 


  Matrix4 r = {{t * x*x + c,     t * x*y + s*z,   t * x*z - s*y, 0,


                t * x*y - s*z,   t * y*y + c,     t * y*z + s*x, 0,


                t * x*z + s*y,   t * y*z - s*x,   t * z*z + c,   0,


                0,               0,               0,             1}};


  return r;


}


 


Matrix4 matrix4_rotation(Quaternion q) // FORWARD DECLARED


{


  float x2 = 2 * q.value.x;


  float y2 = 2 * q.value.y;


  float z2 = 2 * q.value.z;


  float xx2 = q.value.x * x2;


  float yy2 = q.value.y * y2;


  float zz2 = q.value.z * z2;


  float xy2 = q.value.x * y2;


  float xz2 = q.value.x * z2;


  float yz2 = q.value.y * z2;


  float wx2 = q.value.w * x2;


  float wy2 = q.value.w * y2;


  float wz2 = q.value.w * z2;


 


  Matrix4 r = {{1 - yy2 - xx2,  xy2 + wz2,      xz2 - wy2,     0,


                xy2 - wz2,      1 - xx2 - zz2,  yz2 + wx2,     0,


                xz2 + wy2,      yz2 - wx2,      1 - xx2 - yy2, 0,


                0,              0,              0,             1}};


  return r;


}


 


Matrix4 matrix4_scale(float x, float y, float z)


{


  Matrix4 r = {{x, 0, 0, 0,


                0, y, 0, 0,


                0, 0, z, 0,


                0, 0, 0, 1}};


  return r;


}


 


Matrix4 matrix4_trs(Vector3 pos, Quaternion q, Vector3 s)


{


  // TODO do this in one step! Right now I'm doing it as 3 matricies to make it simplier!


  Matrix4 r = matrix4_translate(pos.value.x, pos.value.y, pos.value.z);


  r = matrix4_multiply(r, matrix4_rotation(q));


  r = matrix4_multiply(r, matrix4_scale(s.value.x, s.value.y, s.value.z));

}


 


Matrix4 matrix4_perspective(float fov, float aspect, float zNear, float zFar)


{


  // FROM https://github.com/hitchh1k3r/gitFighter/blob/gh-pages/web/src/VectorMath.dart


  float top = zNear * tanf(fov * PI / 360.0);


  float bottom = -top;


  float left = bottom * aspect;


  float right = top * aspect;


 


  float x =         2 * zNear / (right - left);


  float y =         2 * zNear / (top - bottom);


  float a =    (right + left) / (right - left);


  float b =    (top + bottom) / (top - bottom);


  float c =   -(zFar + zNear) / (zFar - zNear);


  float d = -2 * zFar * zNear / (zFar - zNear);


 


  Matrix4 r = {{x,  0,  0,  0,


                0,  y,  0,  0,


                a,  b,  c, -1,


                0,  0,  d,  0}};


  return r;


}


 


Matrix4 matrix4_orthogonal(float xLeft, float xRight,


                           float yTop, float yBottom,


                           float zNear, float zFar)


{


  // FROM https://github.com/hitchh1k3r/gitFighter/blob/gh-pages/web/src/VectorMath.dart


  // TODO Test this method!


  float invWidth = 1.0 / (xRight - xLeft);


  float invHeight = 1.0 / (yTop - yBottom);


  float invDepth = 1.0 / (zNear - zFar);


 


  Matrix4 r =


        {{2*invWidth,                 0,                           0,                       0,


          0,                          2*invHeight,                 0,                       0,


          0,                          0,                           2*invDepth,              0,


         -1*(xLeft+xRight)*invWidth, -1*(yBottom+yTop)*invHeight, -1*(zFar+zNear)*invDepth, 1}};


  return r;


}
