using System;


using System.IO;


using System.Globalization;


using System.Security.Cryptography;


using UnityEngine;


using UnityEditor;


 


namespace HitchLib


{


 


  [InitializeOnLoad]


  public class AutoSaveSystem


  {


 


    private const string SETTINGS_PATH = "Assets/HitchLib/Auto Save/Editor/settings.asset";


    private static AutoSaveSettings settings;


    private static DateTime lastSaveTime = DateTime.Now;


 


    static AutoSaveSystem()


    {


      settings = GetSettings();


      lastSaveTime = DateTime.FromBinary(settings.lastSceneTime);


      EditorApplication.update += Update;


    }


 


    private static string CalculateHash(string filename)


    {


      /* Calculates MD5 hash of the specified file. */


 


      using(HashAlgorithm md5 = MD5.Create())


      {


          using(FileStream stream = File.OpenRead(filename))


          {


              byte[] hash = md5.ComputeHash(stream);


              return BitConverter.ToString(hash).Replace("-", "").ToLowerInvariant();


          }


      }


    }


 


    public static AutoSaveSettings GetSettings()


    {


      /* Gets or creates a settings file for this system. */


 


      AutoSaveSettings retSettings;


 


      if(settings != null)


      {


        retSettings = settings;


      }


      else


      {


        retSettings = AssetDatabase.LoadAssetAtPath<AutoSaveSettings>(SETTINGS_PATH);


        if(retSettings == null)


        {


          retSettings = ScriptableObject.CreateInstance<AutoSaveSettings>();


          AssetDatabase.CreateAsset(retSettings, SETTINGS_PATH);


        }


      }


 


      return retSettings;


    }


 


    private static void Update()


    {


      /* If the specified interval has elapsed, we save the scene (and then delete it if no changes


         were made -- to save disk space ^_^). */


 


      if(settings == null)


      {


        settings = GetSettings();


      }


 


      if(settings.sceneSave)


      {


        double interval = settings.sceneInterval;


        if((System.DateTime.Now - lastSaveTime).TotalSeconds > 60 * interval)


        {


          if(!string.IsNullOrEmpty(EditorApplication.currentScene) &&


                File.Exists(EditorApplication.currentScene))


          {


            DateTime lastSave = File.GetLastWriteTime(EditorApplication.currentScene);


            if(lastSave > lastSaveTime)


            {


              lastSaveTime = lastSave;


              settings.lastSceneTime = lastSaveTime.ToBinary();


              return;


            }


          }


 


          if(EditorApplication.isSceneDirty)


          {


            lastSaveTime = System.DateTime.Now;


            settings.lastSceneTime = lastSaveTime.ToBinary();


            string path = Path.Combine(Application.dataPath, settings.sceneFolder);


 


            if(!Directory.Exists(path))


            {


              Directory.CreateDirectory(path);


            }


 


            string currentScene = EditorApplication.currentScene;


            if(string.IsNullOrEmpty(currentScene))


            {


              currentScene = "Untitled";


            }


            string dateTime = DateTime.Now.ToString("yyyy-MM-dd_HH-mm-ss",


                  CultureInfo.InvariantCulture);


            string newName = string.Format("{0}_{1}.unity",


                  Path.GetFileNameWithoutExtension(currentScene), dateTime);


 


            string folder = Path.Combine("Assets", settings.sceneFolder);


            string saveFile = Path.Combine(folder, newName);


 


            EditorApplication.SaveScene(saveFile, true);


            string saveHash = CalculateHash(saveFile);


            if(saveHash.Equals(settings.lastSceneHash))


            {


              // The scene has not changed!


              AssetDatabase.DeleteAsset(saveFile);


            }


            else if(settings.sceneLogSave)


            {


              Debug.Log("Scene \"" + currentScene + "\" AutoSaved (" + dateTime + ")");


            }


            settings.lastSceneHash = saveHash;


            AssetDatabase.SaveAssets();


          }


          else


          {


            lastSaveTime = System.DateTime.Now.AddMinutes((interval > 1.5) ? -1 : (-0.5*interval));


            settings.lastSceneTime = lastSaveTime.ToBinary();


          }


        }


      }


    }


 


  }


 


}
