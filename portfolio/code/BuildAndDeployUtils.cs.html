using System;


using System.Reflection;


using System.Collections;


using System.Collections.Generic;


using UnityEngine;


using UnityEditor;


 


namespace HitchLib


{


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  public static class BuildAndDeployUtils


  {


    /* Static class to store all our utilities. */


 


    // Events:


    public delegate void BuildDelegate(BuildDef build, BuildAndDeploySettings settings);


    public delegate void DeployDelegate(BuildDef build, Deployment deployment,


          BuildAndDeploySettings settings);


 


    public static event BuildDelegate PreBuild, PostBuild, CancelBuild;


    public static event DeployDelegate PreDeploy, PostDeploy, CancelDeploy;


 


    // Interface Data:


    private static string[] deploymentTargets;


    private static IDeploymentScript[] deploymentScripts;


    private static MonoScript[] deploymentAssets;


    private static IBuildDeployConfig[] configScripts;


 


    // Methods To Invoke Events:


    public static void InvokePreBuild(BuildDef build, BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      PreBuild(build, settings);


    }


 


    public static void InvokePostBuild(BuildDef build, BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      PostBuild(build, settings);


    }


 


    public static void InvokeCancelBuild(BuildDef build, BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      CancelBuild(build, settings);


    }


 


    public static void InvokePreDeploy(BuildDef build, Deployment deployment,


          BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      PreDeploy(build, deployment, settings);


    }


 


    public static void InvokePostDeploy(BuildDef build, Deployment deployment,


          BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      PostDeploy(build, deployment, settings);


    }


 


    public static void InvokeCancelDeploy(BuildDef build, Deployment deployment,


          BuildAndDeploySettings settings)


    {


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      CancelDeploy(build, deployment, settings);


    }


 


    // Methods For Configuration (De)Serialization:


    public static string UserSettingsKey()


    {


      return "BuildAndDeploy_UserSettings_" + PlayerSettings.productGUID;


    }


 


    public static IBuildDeployConfig[] GetConfigurableScripts()


    {


      /* Returns a list of all configurable script interfaces. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


 


      return configScripts;


    }


 


    // Methods For Deployments Types:


    public static string[] GetDeploymentNames()


    {


      /* Returns a list of all deployment names. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


      return deploymentTargets;


    }


 


    public static IDeploymentScript GetDeploymentScript(int id)


    {


      /* Return the deployment script by numeric id. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


      if(id >= 0 && id < deploymentScripts.Length)


      {


        return deploymentScripts[id];


      }


      return null;


    }


 


    public static MonoScript GetScriptFile(int id)


    {


      /* Return the deployment script's asset file by numeric id. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


      if(id >= 0 && id < deploymentAssets.Length)


      {


        return deploymentAssets[id];


      }


      return null;


    }


 


    public static int GetDeploymentID(string name)


    {


      /* Return the numeric id of the deployment by it's name. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


      for(int i = 0; i < deploymentTargets.Length; ++i)


      {


        if(deploymentTargets[i] == name)


        {


          return i;


        }


      }


      return -1;


    }


 


    public static string GetDeploymentName(int id)


    {


      /* Return the name of the deployment by numeric id. */


 


      if(AssetUpdates.dirty)


      {


        UpdateInterfaces();


      }


      if(id >= 0 && id < deploymentTargets.Length)


      {


        return deploymentTargets[id];


      }


      return "";


    }


 


    private static void UpdateInterfaces()


    {


      /* If we use the utility methods after any assets have changed, we will reload our interface


         data so we are never outdated. It's a bit heavy, but it's only done when needed, and


         applicable. */


 


      List<string> newTargets = new List<string>();


      List<IDeploymentScript> newScripts = new List<IDeploymentScript>();


      List<MonoScript> newAssets = new List<MonoScript>();


      List<IBuildDeployConfig> newConfigs = new List<IBuildDeployConfig>();


 


      // NOTE (hitch) Activator.CreateInstance can throw an error if there is no accessable default


      // constructor. So we empty our data stores, so that any errors will disabled functionallity


      // until the error is fixed.


 


      // Reset all data storage:


      deploymentTargets = new string[0];


      deploymentScripts = new IDeploymentScript[0];


      deploymentAssets = new MonoScript[0];


      configScripts = new IBuildDeployConfig[0];


      PreBuild = null;


      PostBuild = null;


      CancelBuild = null;


      PreDeploy = null;


      PostDeploy = null;


      CancelDeploy = null;


 


      // Load config options JSON from the settings file:


      BuildAndDeploySettings.JSON_config loadData = new BuildAndDeploySettings.JSON_config();


      EditorJsonUtility.FromJsonOverwrite(BuildAndDeployWindow.GetSettings().configurationJSON,


            loadData);


      BuildAndDeploySettings.JSON_config userData = new BuildAndDeploySettings.JSON_config();


      EditorJsonUtility.FromJsonOverwrite(EditorPrefs.GetString(UserSettingsKey()), userData);


 


      // Search scripts for our interfaces:


      string[] guids = AssetDatabase.FindAssets("t:MonoScript");


      foreach(string guid in guids)


      {


        MonoScript script = AssetDatabase.LoadAssetAtPath<MonoScript>(


              AssetDatabase.GUIDToAssetPath(guid));


 


        if(script.GetClass() != null)


        {


          Type[] interfaces = script.GetClass().GetInterfaces();


          System.Object instance = null;


 


          if(ArrayUtility.Contains<Type>(interfaces, typeof(IDeploymentScript)))


          {


            if(instance == null)


            {


              instance = Activator.CreateInstance(script.GetClass());


            }


 


            newAssets.Add(script);


            newScripts.Add((IDeploymentScript) instance);


            newTargets.Add(((IDeploymentScript) instance).GetID());


          }


 


          if(ArrayUtility.Contains<Type>(interfaces, typeof(IBuildDeployConfig)))


          {


            if(instance == null)


            {


              instance = Activator.CreateInstance(script.GetClass());


            }


 


            IBuildDeployConfig conf = (IBuildDeployConfig) instance;


 


            newConfigs.Add(conf);


 


            BuildAndDeployUtils.ConfigOption[] options = conf.GetOptions();


 


            // Find the corresponding JSON (project settings) options for this configuration:


            if(loadData != null && loadData.elements != null)


            {


              foreach(BuildAndDeploySettings.JSON_config_element element in loadData.elements)


              {


                if(conf.GetStorageID() == element.name)


                {


                  // Save the options from the JSON object into the configuration script.


                  for(int oi = 0; oi < options.Length; ++oi)


                  {


                    if(!options[oi].userSetting)


                    {


                      for(int ji = 0; ji < element.options.Length; ++ji)


                      {


                        if(options[oi].name == element.options[ji].name)


                        {


                          options[oi] = element.options[ji];


                        }


                      }


                    }


                  }


                }


              }


            }


 


            // Find the corresponding JSON (user settings) options for this configuration:


            if(userData != null && userData.elements != null)


            {


              foreach(BuildAndDeploySettings.JSON_config_element element in userData.elements)


              {


                if(conf.GetStorageID() == element.name)


                {


                  // Save the options from the JSON object into the configuration script.


                  for(int oi = 0; oi < options.Length; ++oi)


                  {


                    if(options[oi].userSetting)


                    {


                      for(int ji = 0; ji < element.options.Length; ++ji)


                      {


                        if(options[oi].name == element.options[ji].name)


                        {


                          options[oi] = element.options[ji];


                        }


                      }


                    }


                  }


                }


              }


            }


 


            conf.SetConfig(options);


          }


 


          if(ArrayUtility.Contains<Type>(interfaces, typeof(IBuildDeployEventListener)))


          {


            if(instance == null)


            {


              instance = Activator.CreateInstance(script.GetClass());


            }


 


            PreBuild += ((IBuildDeployEventListener) instance).PreBuild;


            PostBuild += ((IBuildDeployEventListener) instance).PostBuild;


            CancelBuild += ((IBuildDeployEventListener) instance).CancelBuild;


            PreDeploy += ((IBuildDeployEventListener) instance).PreDeploy;


            PostDeploy += ((IBuildDeployEventListener) instance).PostDeploy;


            CancelDeploy += ((IBuildDeployEventListener) instance).CancelDeploy;


          }


        }


      }


 


      // Store new values:


      deploymentTargets = newTargets.ToArray();


      deploymentScripts = newScripts.ToArray();


      deploymentAssets = newAssets.ToArray();


      configScripts = newConfigs.ToArray();


 


      AssetUpdates.dirty = false;


    }


 


    // HitchLib // BuildAndDeployUtils //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*/


    [Serializable]


    public struct Configuration


    {


 


      public string name;


      public ConfigOption[] options;


 


    }


 


    // HitchLib // BuildAndDeployUtils //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*/


    [Serializable]


    public struct ConfigOption


    {


 


      public string name;


      public bool userSetting;


      public ConfigType type;


      public int iVal;


      public float fVal;


      public bool bVal;


      public string sVal;


      public Type rType;


      public UnityEngine.Object rVal;


 


    }


 


    // HitchLib // BuildAndDeployUtils //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*/


    public enum ConfigType


    { INT, FLOAT, STRING, OBJREF, BOOL, ENUM }


 


    // HitchLib // BuildAndDeployUtils //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*/


    public class AssetUpdates : AssetPostprocessor


    {


      /* Internal class to listen for asset update messages from the Unity Editor. This way utility


         methods can tell if they need to update their cached interface lists and data. */


 


      public static bool dirty = true;


 


      static void OnPostprocessAllAssets(string[] i, string[] d, string[] t, string[] f)


      {


        /* On any changes set the dirty flag on. */


 


        dirty = true;


      }


 


    }


 


  }


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  public interface IDeploymentScript


  {


    /* Interface for deployment scripts. This allows us to program nicely modular deployment. */


 


    string GetID();


    string GetDisplayName();


    IEnumerator DoDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings,


          BuildAndDeployWindow window);


    bool RecoverFromError(BuildDef build, Deployment deployment, BuildAndDeploySettings settings,


          System.Diagnostics.Process proc, BuildAndDeployWindow window);


 


  }


 


  // HitchLib //*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//*//


  public interface IBuildDeployConfig


  {


    /* Interface for any script that should be able to serialize it's state in the settings file. */


 


    string GetStorageID();


    BuildAndDeployUtils.ConfigOption[] GetOptions();


    void SetConfig(BuildAndDeployUtils.ConfigOption[] config);


 


  }

  public interface IBuildDeployEventListener


  {


    /* Interface for any script that should be told before and after each build and deployment. */


 


    void PreBuild(BuildDef build, BuildAndDeploySettings settings);


    void PostBuild(BuildDef build, BuildAndDeploySettings settings);


    void CancelBuild(BuildDef build, BuildAndDeploySettings settings);


    void PreDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings);


    void PostDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings);


    void CancelDeploy(BuildDef build, Deployment deployment, BuildAndDeploySettings settings);


 


  }
